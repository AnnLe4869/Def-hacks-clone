interface User {
  id: User_ID;
  display_name: string;
  email: string;
  progress: [
    {
      id: Lesson_ID;
      dateComplete: string;
    }
  ];
  lastLesson: Lesson_ID;
}

interface Course {
  id: Course_ID;
  course_name: string;
  content: [{ id: string; lessonName: Lesson_ID }];
  // Ordinal number, indicate the position in line of the course, as one course must followed by another specific course
  no: number;
}

interface Lesson {
  id: Lesson_ID;
  lessonName: string;
  quizzes: [
    {
      id: Quiz_ID;
      options: [{ option_id: Option_ID; optionContent: string }];
      correctOption: Option_ID;
    }
  ];
  instruction: string;
  practice: {
    code: string;
    test: string;
  };
  // Ordinal number, indicate the position in line of the lesson, as one lesson must followed by another specific lesson
  no: number;
  // Path to file to download
  files: string[];
}

type User_ID = string;
type Course_ID = string;
type Lesson_ID = string;
type Quiz_ID = string;
type Option_ID = string;
